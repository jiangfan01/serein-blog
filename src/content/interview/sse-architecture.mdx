### 1. API 层 (services/stream/agentStreamApi.ts)

**职责**: 与后端建立 SSE 连接

**核心方法**:

```typescript
// 发送消息并建立 SSE 连接
export const sendAgentStreamRequest = async (data: AgentStreamRequest): Promise<Response> => {
  const response = await fetch(`${baseURL}/agent_adk/chat/send`, {
    method: "POST",
    headers: {
      "Content-Type": "application/json",
      Accept: "text/event-stream", // ⭐ 关键：指定接收 SSE 流
    },
    credentials: "include",
    body: JSON.stringify({ ...data, stream: true }),
  });
  return response; // 返回 Response 对象，包含 body: ReadableStream
};

// 订阅正在执行的流（断线重连）
export const subscribeAgentExecution = async (executionId: string): Promise<Response> => {
  const response = await fetch(`${baseURL}/agent_adk/chat/executions/${executionId}/subscribe`, {
    method: "GET",
    headers: { Accept: "text/event-stream" },
    credentials: "include",
  });
  return response;
};
```

**关键点**:
- 必须使用 `fetch` 而非 `axios`（axios 不支持流式响应）
- `Accept: text/event-stream` 告诉服务器返回 SSE 格式
- 返回的 `Response.body` 是 `ReadableStream<Uint8Array>`

### 2. SSE 解析层 (utils/stream/sseParser.ts)

**职责**: 解析 SSE 协议格式

**SSE 协议格式**:
```
event: message
data: {"content": "Hello"}

event: tool_call
data: {"tool_name": "search"}

event: complete
data: {}
```

**核心方法**:

```typescript
export async function* parseSSEStream(options: StreamOptions): AsyncGenerator<SSEEvent> {
  const decoder = new TextDecoder("utf-8");
  const reader = readableStream.getReader();
  let buffer = "";

  while (true) {
    const { done, value } = await reader.read();
    if (done) break;

    buffer += decoder.decode(value, { stream: true });

    // 处理完整的事件 (以 \n\n 分隔)
    let newlineIndex;
    while ((newlineIndex = buffer.indexOf("\n\n")) !== -1) {
      const chunk = buffer.slice(0, newlineIndex);
      buffer = buffer.slice(newlineIndex + 2);

      const event = processBuffer(chunk + "\n\n", currentEvent);
      if (event && (event.data || event.event)) {
        yield event; // ⭐ 生成器模式，逐个返回事件
      }
    }
  }
}
```

**关键点**:
- 使用 `AsyncGenerator` 实现流式迭代
- 按 `\n\n` 分隔事件
- 支持 `event`、`data`、`id`、`retry` 字段

### 3. 流处理层 (services/stream/agentStreamService.ts)

**职责**: 处理 SSE 流，解析不同事件类型

**核心方法**:

```typescript
export async function processSSEStream(
  readableStream: ReadableStream<Uint8Array>,
  options: SSEProcessorOptions
): Promise<void> {
  const { handlers, handleSessionInfo = false, debug = false } = options;
  const state = createSSEProcessorState();

  // 遍历 SSE 流
  for await (const chunk of createSSEStream({ readableStream })) {
    const { event: eventType = "message", data } = chunk;

    // 根据事件类型分发处理
    switch (eventType) {
      case "message":
        // 处理普通消息 + 流式 JSON 解析
        const result = state.streamingJsonParser.feed(content);
        if (result.plainText) {
          addEventToBuffer({ type: StreamEventType.MESSAGE, content: result.plainText });
        }
        if (result.inTypedMode) {
          addEventToBuffer({ type: StreamEventType.TYPED_MESSAGE, toolData: result.partialData });
        }
        break;

      case "tool_call":
        const event = parseToolCallEvent(data);
        addEventToBuffer(event);
        break;

      case "complete":
        handlers.onEvent(parseCompleteEvent());
        break;

      // ... 其他事件类型
    }
  }
}
```

**支持的事件类型**:
| SSE 事件类型 | StreamEventType | 说明 |
|-------------|----------------|------|
| `message` | MESSAGE | 普通文本消息 |
| `message` (含 `<typed>`) | TYPED_MESSAGE | 结构化消息（如图表、卡片） |
| `reasoning` | REASONING | 深度思考内容 |
| `steps` | STEPS | 执行计划步骤 |
| `tool_call` | TOOL_CALL | 工具调用开始 |
| `tool_call_end` | TOOL_END | 工具调用结束 |
| `step_message` | STEP_MESSAGE | 步骤内消息 |
| `step_tool_call` | STEP_TOOL_CALL | 步骤内工具调用 |
| `complete` | COMPLETE | 流完成 |
| `error` | ERROR | 错误 |
| `paused` | PAUSED | 用户暂停 |

**流式 JSON 解析器**:

```typescript
// 检测并解析 <typed type="chart">{"data": ...}</typed> 格式
class StreamingJsonParser {
  feed(chunk: string) {
    this.buffer += chunk;

    // 检测 <typed type="xxx"> 开头
    if (!this.inTypedMode && this.buffer.includes("<typed type=")) {
      this.inTypedMode = true;
      this.type = extractType(this.buffer);
    }

    // 在 typed 模式下，持续收集 JSON 数据
    if (this.inTypedMode) {
      const jsonContent = extractJsonBetweenTags(this.buffer);
      this.partialData = parsePartialJson(jsonContent); // 增量解析
    }

    // 检测 </typed> 结束
    if (this.buffer.includes("</typed>")) {
      this.isComplete = true;
    }

    return {
      plainText: extractPlainText(this.buffer),
      inTypedMode: this.inTypedMode,
      type: this.type,
      partialData: this.partialData,
      isComplete: this.isComplete,
    };
  }
}
```

### 4. 单例管理层 (services/stream/globalSSEManager.ts) ⭐

**职责**: 全局 SSE 连接管理，独立于 React 生命周期

**设计模式**: 单例模式

**核心功能**:

```typescript
class GlobalSSEManager {
  private static instance: GlobalSSEManager;
  private connections: Map<string, SSEConnectionInfo> = new Map();
  private eventCallback: SSEEventCallback | null = null;
  private readonly MAX_CONNECTIONS = 10; // 最大并发连接数
  private readonly MAX_RETRY = 3; // 最大重试次数

  // 单例模式
  static getInstance(): GlobalSSEManager {
    if (!GlobalSSEManager.instance) {
      GlobalSSEManager.instance = new GlobalSSEManager();
    }
    return GlobalSSEManager.instance;
  }

  // 创建 SSE 连接
  createConnection(sessionId: string, executionId: string, source: ConnectionSource): boolean {
    // 1. 检查是否已有活跃连接
    if (this.hasActiveConnection(sessionId)) return false;

    // 2. 检查并发上限
    if (this.getActiveConnectionCount() >= this.MAX_CONNECTIONS) {
      this.eventCallback?.(sessionId, { type: StreamEventType.ERROR, content: "已达到最大并发会话数" });
      return false;
    }

    // 3. 创建连接信息
    const connectionInfo: SSEConnectionInfo = {
      sessionId,
      executionId,
      status: "connecting",
      source,
      retryCount: 0,
      createdAt: Date.now(),
      abortController: new AbortController(),
    };

    this.connections.set(sessionId, connectionInfo);

    // 4. 异步建立连接
    this.establishConnection(sessionId, executionId);
    return true;
  }

  // 建立 SSE 连接
  private async establishConnection(sessionId: string, executionId: string): Promise<void> {
    try {
      const response = await subscribeAgentExecution(executionId);
      if (!response.body) {
        this.handleConnectionComplete(sessionId);
        return;
      }

      connectionInfo.status = "connected";

      // 处理 SSE 流
      await processSSEStream(response.body, {
        handlers: {
          onEvent: (event) => this.handleSSEEvent(sessionId, event),
          onError: (error) => this.handleConnectionError(sessionId, error),
        },
      });

      this.handleConnectionComplete(sessionId);
    } catch (error) {
      this.handleConnectionError(sessionId, error.message);
    }
  }

  // 处理 SSE 事件
  private handleSSEEvent(sessionId: string, event: StreamEvent): void {
    // 调用回调，将事件传递给 Store
    this.eventCallback?.(sessionId, event);

    // 检查是否是结束事件
    if (event.type === StreamEventType.COMPLETE || event.type === StreamEventType.ERROR) {
      this.handleConnectionComplete(sessionId);
    }
  }

  // 处理连接错误（包含重试逻辑）
  private handleConnectionError(sessionId: string, error: string): void {
    const connectionInfo = this.connections.get(sessionId);
    connectionInfo.retryCount++;

    if (connectionInfo.retryCount < this.MAX_RETRY) {
      const retryDelay = 1000 * connectionInfo.retryCount;
      setTimeout(() => {
        if (this.connections.has(sessionId)) {
          this.establishConnection(sessionId, connectionInfo.executionId);
        }
      }, retryDelay);
    } else {
      // 重试耗尽，发送错误事件
      this.eventCallback?.(sessionId, {
        type: StreamEventType.ERROR,
        content: `连接失败: ${error}`,
      });
      this.connections.delete(sessionId);
    }
  }

  // 关闭连接
  closeConnection(sessionId: string): void {
    const connectionInfo = this.connections.get(sessionId);
    if (connectionInfo) {
      connectionInfo.abortController?.abort();
      connectionInfo.status = "disconnected";
    }
    this.connections.delete(sessionId);
  }
}

// 导出单例实例
export const globalSSEManager = GlobalSSEManager.getInstance();
```

**关键特性**:
1. **单例模式**: 全局唯一实例，独立于 React 组件生命周期
2. **连接池管理**: 最多 10 个并发连接
3. **自动重试**: 失败后最多重试 3 次，间隔递增（1s, 2s, 3s）
4. **事件回调**: 通过回调将事件分发到 Store
5. **生命周期管理**: 自动清理完成/错误的连接

```typescript
function scheduleEventBatch(sessionId: string, event: StreamEvent, processCallback: () => void) {
  const buffer = eventBuffers.get(sessionId) || [];
  buffer.push(event);
  eventBuffers.set(sessionId, buffer);

  if (pendingUpdates.has(sessionId)) return;

  });

  pendingUpdates.set(sessionId, frameId);
```

### 6. 事件处理层 (utils/stream/streamEventProcessor.ts) ⭐

**职责**: 将原始 `StreamEvent[]` 转换为渲染友好的 `ProcessedEvent[]`

**核心逻辑**:

```typescript
export const processStreamEvents = (events: StreamEvent[]): ProcessedEvent[] => {
  const processed: ProcessedEvent[] = [];

  // 1. 先处理 steps 事件 → 合并所有步骤状态
  const allStepsEvents = events.filter((e) => e.type === StreamEventType.STEPS);
  const stepsMap = new Map<number, StepItem>();

  allStepsEvents.forEach((event) => {
    const { type, steps, step_id, status } = event.toolData || {};
    if (type === "init" && steps) {
      steps.forEach((step) => stepsMap.set(step.id, step));
    } else if (type === "update") {
      const id = step_id ?? stepId;
      if (stepsMap.has(id)) {
        stepsMap.set(id, { ...stepsMap.get(id), status });
      }
    }
  });

  const mergedSteps = Array.from(stepsMap.values()).sort((a, b) => a.id - b.id);
  if (mergedSteps.length > 0) {
    processed.push({
      type: "steps",
      toolData: { type: "init", steps: mergedSteps },
      originalEvents: allStepsEvents,
    });
  }

  // 2. 处理 step_message 和 step_tool_call → 按 stepId 分组
  const stepGroups = new Map<number, Array<{ type: "message" | "tool"; content?: string; toolData?: any }>>();

  stepMessageEvents.forEach((event) => {
    const stepId = getCurrentStepId(allStepsEvents, event.timestamp);
    if (stepId && event.content) {
      if (!stepGroups.has(stepId)) stepGroups.set(stepId, []);
      stepGroups.get(stepId)!.push({ type: "message", content: event.content });
    }
  });

  stepToolEvents.forEach((event) => {
    const stepId = getCurrentStepId(allStepsEvents, event.timestamp);
    if (stepId) {
      if (!stepGroups.has(stepId)) stepGroups.set(stepId, []);
      stepGroups.get(stepId)!.push({ type: "tool", toolData: event.toolData });
    }
  });

  // 为每个步骤创建 step_group
  stepGroups.forEach((items, stepId) => {
    const stepTitle = stepTitles.get(stepId);
    if (stepTitle && items.length > 0) {
      processed.push({
        type: "step_group",
        stepTitle,
        toolData: { orderedContent: items, stepId },
        originalEvents: [{ id: `step-group-${stepId}` }],
      });
    }
  });

  // 3. 处理其他事件（message, tool, reasoning 等）
  let i = 0;
  while (i < sortedEvents.length) {
    const event = sortedEvents[i];

    switch (event.type) {
      case StreamEventType.MESSAGE:
        // 合并连续的 message 事件
        const messageEvents = [event];
        let j = i + 1;
        while (j < sortedEvents.length && sortedEvents[j].type === StreamEventType.MESSAGE) {
          messageEvents.push(sortedEvents[j]);
          j++;
        }
        const combinedContent = messageEvents.map((e) => e.content || "").join("").trim();
        processed.push({ type: "message", content: combinedContent, originalEvents: messageEvents });
        i = j;
        break;

      case StreamEventType.TOOL_CALL:
        // 查找对应的 TOOL_END 事件
        const endEvent = sortedEvents.find((e) => e.type === StreamEventType.TOOL_END && e.id === event.id);
        processed.push({
          type: "tool",
          toolData: endEvent ? { ...event.toolData, ...endEvent.toolData } : event.toolData,
          isComplete: !!endEvent,
          originalEvents: endEvent ? [event, endEvent] : [event],
        });
        i++;
        break;

      case StreamEventType.TYPED_MESSAGE:
        // 使用 id 去重，只保留最新的 typed_message 事件
        const existingIndex = processed.findIndex((p) => p.type === "typed_message" && p.originalEvents[0]?.id === event.id);
        const typedMessageProcessed = {
          type: "typed_message",
          messageType: event.toolData?.messageType,
          messageData: event.toolData?.messageData,
          isComplete: event.toolData?.isComplete,
          originalEvents: [event],
        };
        if (existingIndex >= 0) {
          processed[existingIndex] = typedMessageProcessed; // 更新
        } else {
          processed.push(typedMessageProcessed); // 添加
        }
        i++;
        break;

      // ... 其他事件类型
    }
  }

  return processed;
};
```

**处理策略**:
1. **steps 事件**: 合并所有步骤状态，生成一个 "steps" ProcessedEvent
2. **step_group 事件**: 按 stepId 分组，生成多个 "step_group" ProcessedEvent
3. **message 事件**: 合并连续的消息
4. **tool 事件**: 配对 TOOL_CALL 和 TOOL_END
5. **typed_message 事件**: 使用 id 去重，支持流式更新

```typescript
class GlobalSSEManager {
```

## ⚡ 性能优化

### 1. 批量事件处理

**问题**: SSE 事件频繁触发，导致频繁的 React 渲染

**解决方案**: 使用 `requestAnimationFrame` 批量处理事件

```typescript
function scheduleEventBatch(sessionId: string, event: StreamEvent, processCallback: () => void) {
  // 将事件添加到缓冲区
  const buffer = eventBuffers.get(sessionId) || [];
  buffer.push(event);
  eventBuffers.set(sessionId, buffer);

  // 如果已有待处理的更新，跳过
  if (pendingUpdates.has(sessionId)) return;

  // 使用 requestAnimationFrame 批量处理
  const frameId = requestAnimationFrame(() => {
    pendingUpdates.delete(sessionId);
    processCallback();
  });

  pendingUpdates.set(sessionId, frameId);
}
```

**效果**: 将多个事件合并为一次渲染，减少渲染次数

### 2. RAF 优化渲染

**问题**: `processedEvents` 频繁更新，导致组件频繁重渲染

**解决方案**: 在渲染层使用 `requestAnimationFrame` 优化

```typescript
const [optimizedEvents, setOptimizedEvents] = React.useState(processedEvents);
const rafIdRef = React.useRef<number | null>(null);

React.useEffect(() => {
  if (rafIdRef.current) cancelAnimationFrame(rafIdRef.current);

  rafIdRef.current = requestAnimationFrame(() => {
    setOptimizedEvents(processedEvents);
    rafIdRef.current = null;
  });

  return () => {
    if (rafIdRef.current) cancelAnimationFrame(rafIdRef.current);
  };
}, [processedEvents]);
```

**效果**: 将高频更新限制在浏览器刷新率（60fps）

**问题**: 父组件更新导致子组件不必要的重渲染

### 5. LRU 缓存清理

**问题**: 会话数量过多，占用内存

**解决方案**: 使用 LRU 策略清理旧会话

```typescript
cleanupOldSessions: () => {
  const MAX_CACHED_SESSIONS = 20;

  // 按访问时间排序
  const sortedSessions = Array.from(state.sessions.entries())
    .sort(([, a], [, b]) => b.lastAccessTime - a.lastAccessTime);

  // 保留最近访问的会话
  const newSessions = new Map<string, SessionState>();
  let kept = 0;

  for (const [sessionId, session] of sortedSessions) {
    const hasActiveConnection = globalSSEManager.hasActiveConnection(sessionId);
    const isCurrentSession = sessionId === state.currentViewingSessionId;

    if (kept < MAX_CACHED_SESSIONS || hasActiveConnection || isCurrentSession) {
      newSessions.set(sessionId, session);
      kept++;
    } else {
      globalSSEManager.closeConnection(sessionId);
    }
  }

  return { sessions: newSessions };
}
```

### 1. 连接错误处理

**GlobalSSEManager 自动重试**:

```typescript
private handleConnectionError(sessionId: string, error: string): void {
  const connectionInfo = this.connections.get(sessionId);
  connectionInfo.retryCount++;

  if (connectionInfo.retryCount < this.MAX_RETRY) {
    // 重试，间隔递增（1s, 2s, 3s）
    const retryDelay = 1000 * connectionInfo.retryCount;
    setTimeout(() => {
      if (this.connections.has(sessionId)) {
        this.establishConnection(sessionId, connectionInfo.executionId);
      }
    }, retryDelay);
  } else {
    // 重试耗尽，发送错误事件
    this.eventCallback?.(sessionId, {
      type: StreamEventType.ERROR,
      content: `连接失败: ${error}`,
    });
    this.connections.delete(sessionId);
  }
}
```
